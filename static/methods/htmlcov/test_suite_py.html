import unittest 

 

from DuplicateProcessing import DuplicateProcessing 

from PreProcessing import PreProcessing 

from SimilarityProcessing import SimilarityProcessing 

 

 

class Tests(unittest.TestCase): 

 

# Tests for PreProcessing 

 

# Tests to make sure that true case takes a string and makes all characters lowercase, tests several scenarios. 

def test_true_case(self): 

prep = PreProcessing() 

test_string1 = "UPPERCASE" 

self.assertEquals(prep.true_case(test_string1), "uppercase") 

test_string2 = "MiXeD CAsE" 

self.assertEquals(prep.true_case(test_string2), "mixed case") 

test_string3 = "lower case" 

self.assertEquals(prep.true_case(test_string3), "lower case") 

 

 

# Tests the strip method, makes sure all unecessary characters are removed. 

def test_strip_file(self): 

prep = PreProcessing() 

test_string1 = "hi!@ w.h?y, (I $don't know) #what= these characters; are -doing-here" 

self.assertEquals(prep.strip_file(test_string1), "hi why I don't know what these characters are doing here") 

test_string2 = " uneven spaces are my favorite thing ever!!!!!!" 

self.assertEquals(prep.strip_file(test_string2), " uneven spaces are my favorite thing ever") 

test_string3 = ' "Quote test" ' 

self.assertEquals(prep.strip_file(test_string3), ' Quote test ') 

 

# Tests the stop words method 

def test_stop_words(self): 

prep = PreProcessing() 

test_string1 = "he said that it might not be a good idea" 

self.assertNotEqual(len(prep.stop_words(test_string1)), len(test_string1)) 

 

# Tests the porter stem method 

def test_porter_stem(self): 

prep = PreProcessing() 

test_string1 = "longer words are highly likely to be stemmed how wonderful" 

self.assertNotEqual(prep.porter_stem(test_string1), test_string1) 

 

# Tests for Duplicate Processing 

 

def test_compute_sha224_not_null(self): 

dp = DuplicateProcessing() 

shingle = "test,test,test" 

 

self.assertIsNotNone(dp.compute_sha224(shingle)) 

 

def test_compute_sha224_null(self): 

dp = DuplicateProcessing() 

shingle = None 

 

self.assertEquals(dp.compute_sha224(shingle), "ERROR: Null Shingle Passed") 

 

def test_shingled_roof_dup_empty(self): 

dp = DuplicateProcessing() 

shingle_list = [] 

shingle_set = set() 

 

self.assertEquals(dp.shingled_roof_dup(shingle_list), shingle_set) 

 

def test_shingled_roof_dup_not_empty(self): 

dp = DuplicateProcessing() 

sample_test = ["This", "is", "a", "test", "string", "of", "at", "least", "nine", "words", "or", "ect"] 

 

# Checks that the method breaks up the test string into 1 shingles 

self.assertEquals(len(dp.shingled_roof_dup(sample_test)), 1) 

 

# Tests for Similarity Matching 

 

def test_shingled_roof_sim_empty(self): 

sp = SimilarityProcessing() 

shingle_list = [] 

shingle_set = set() 

 

self.assertEquals(sp.shingled_roof_sim(shingle_list), shingle_set) 

 

def test_shingled_roof_sim_not_empty(self): 

sp = SimilarityProcessing() 

sample_test = ["This", "is", "a", "test", "string", "of", "at", "least", "nine", "words", "or", "ect"] 

 

# Checks that the method breaks up the test string into 10 shingles, allows for repeats 

self.assertEquals(len(sp.shingled_roof_sim(sample_test)), 10) 

 

def test_min_hash(self): 

sp = SimilarityProcessing() 

sample_shingle_list = ["hi there I", "why what now", "oh sure of", "oops stop word"] 

sample_shingle_set = set(sample_shingle_list) 

self.assertEquals(len(sp.min_hash(sample_shingle_set)), 200) 

 

if __name__ == '__main__': 

unittest.main()
