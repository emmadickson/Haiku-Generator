# Implementation of the methods needed for crc64 hash. Based off of cookbook recipe. 

# Returns a 64 bit hash and contains a digest method 

 

 

class crc64(object): 

 

# Constants 

CRCTableh = [0] * 256 

CRCTablel = [0] * 256 

POLY64REVh = 0xd8000000L 

BIT_TOGGLE = 1L << 31L 

 

def __init__(self): pass 

 

# Sets up the tables needed to compute the hash 

@staticmethod 

def init(): 

for i in xrange(256): 

partl = i 

parth = 0L 

for j in xrange(8): 

rflag = partl & 1L 

partl >>= 1L 

if parth & 1: 

partl ^= crc64.BIT_TOGGLE 

parth >>= 1L 

if rflag: 

parth ^= crc64.POLY64REVh 

crc64.CRCTableh[i] = parth 

crc64.CRCTablel[i] = partl 

 

# Computes a 64 bit hash number separated into two numbers the upper 32 bit half and 

# the lower 32 bit half. 

@staticmethod 

def crc64_compute(bytes, (crch, crcl)=(0, 0)): 

for byte in bytes: 

shr = (crch & 0xFF) << 24 

temp1h = crch >> 8L 

temp1l = (crcl >> 8L) | shr 

tableindex = (crcl ^ ord(byte)) & 0xFF 

crch = temp1h ^ crc64.CRCTableh[tableindex] 

crcl = temp1l ^ crc64.CRCTablel[tableindex] 

return crch, crcl 

 

# Computes the digest for crc64 

@staticmethod 

def crc64digest(input_string): 

return "" \ 

"%08X%08X" % (crc64.crc64_compute(input_string))
