# Strips the file of un-needed characters like punctuation, ect. 

# TODO: Keep an eye on this one, add or subtract punctuation as needed. 

@staticmethod 

def strip_file(clothed_file): 

print ('Inside the strip_file method') 

stripped_file = re.sub('[!@#$,?;()=.]', '', clothed_file) 

stripped_file = re.sub('[-]', ' ', stripped_file) # Ensures that cases like hi-there become hi there 

stripped_file = re.sub(' +', ' ', stripped_file) # Takes care of uneven spacing 

stripped_file = re.sub('[\n]', ' ', stripped_file) 

stripped_file = stripped_file.replace('"', "") 

print ('Exiting the strip_file method') 

return stripped_file

# Stemming in a utf-8 friendly way with Porter
